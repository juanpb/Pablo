
Function getHoras(asignarA, severidad)   

	Select Case asignarA
		Case "Ambientes Atis" 
			Select Case severidad
				Case "1-Bloqueante"
					hs = 1
				Case "2-Critica"
					hs = 2
				Case "3-Alta"
					hs = 3
				Case "4-Media"
					hs = 4
				Case "5-Baja"
					hs = 5
				End Select
		Case "Ambientes Fija" 
			Select Case severidad
				Case "1-Bloqueante"
					hs = 6
				Case "2-Critica"
					hs = 7
				Case "3-Alta"
					hs = 8
				Case "4-Media"
					hs = 9
				Case "5-Baja"
					hs = 10
				End Select
		Case "Ambientes Movil" 
			Select Case severidad
				Case "1-Bloqueante"
					hs = 11
				Case "2-Critica"
					hs = 12
				Case "3-Alta"
					hs = 13
				Case "4-Media"
					hs = 14
				Case "5-Baja"
					hs = 15
				End Select
		Case Else
           hs = ""
  End Select
	getHoras = hs
End Function
